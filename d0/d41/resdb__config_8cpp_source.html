42ResDBConfig::ResDBConfig(const ResConfigData& config_data,
43 const ReplicaInfo& self_info,
44 const KeyInfo& private_key,
45 const CertificateInfo& public_key_cert_info)
46 : config_data_(config_data),
47 self_info_(self_info),
48 private_key_(private_key),
49 public_key_cert_info_(public_key_cert_info) {
50 for (const auto& region : config_data.region()) {
51 if (region.region_id() == config_data.self_region_id()) {
52 LOG(INFO) << "get region info:" << region.DebugString();
53 for (const auto& replica : region.replica_info()) {
54 replicas_.push_back(replica);
55 }
56 LOG(INFO) << "get region config server size:"
57 << region.replica_info_size();
58 break;
59 }
60 }
61 learners_.clear();
62 for (const auto& learner : config_data_.learner_info()) {
63 learners_.push_back(learner);
64 }
65
66 if (config_data_.block_size() > 0) {
67 block_size_ = config_data_.block_size();
68 }
69 if (config_data_.view_change_timeout_ms() == 0) {
70 config_data_.set_view_change_timeout_ms(viewchange_commit_timeout_ms_);
71 }
72 if (config_data_.client_batch_num() == 0) {
73 config_data_.set_client_batch_num(client_batch_num_);
74 }
75 if (config_data_.worker_num() == 0) {
76 config_data_.set_worker_num(worker_num_);
77 }
78 if (config_data_.input_worker_num() == 0) {
79 config_data_.set_input_worker_num(input_worker_num_);
80 }
81 if (config_data_.output_worker_num() == 0) {
82 config_data_.set_output_worker_num(output_worker_num_);
83 }
84 if (config_data_.tcp_batch_num() == 0) {
85 config_data_.set_tcp_batch_num(100);
86 }
87 if (config_data_.max_process_txn() == 0) {
88 config_data_.set_max_process_txn(64);
89 }
90}

92void ResDBConfig::SetConfigData(const ResConfigData& config_data) {
93 config_data_ = config_data;
94 replicas_.clear();
95 for (const auto& region : config_data.region()) {
96 if (region.region_id() == config_data.self_region_id()) {
97 LOG(INFO) << "get region info:" << region.DebugString();
98 for (const auto& replica : region.replica_info()) {
99 replicas_.push_back(replica);
100 }
101 LOG(INFO) << "get region config server size:"
102 << region.replica_info_size();
103 break;
104 }
105 }
106 learners_.clear();
107 for (const auto& learner : config_data_.learner_info()) {
108 learners_.push_back(learner);
109 }
110 if (config_data_.block_size() > 0) {
111 block_size_ = config_data_.block_size();
112 }
113 if (config_data_.view_change_timeout_ms() == 0) {
114 config_data_.set_view_change_timeout_ms(viewchange_commit_timeout_ms_);
115 }
