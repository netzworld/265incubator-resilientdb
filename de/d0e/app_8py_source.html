98@strawberry.type
99class Mutation:
100 @strawberry.mutation
101 def postTransaction(self, data: PrepareAsset) -> CommitTransaction:
102 prepared_token_tx = db.transactions.prepare(
103 operation=data.operation,
104 signers=data.signerPublicKey,
105 recipients=[([data.recipientPublicKey], data.amount)],
106 asset=data.asset,
107 )
108
109 # fulfill the tnx
110 fulfilled_token_tx = db.transactions.fulfill(prepared_token_tx, private_keys=data.signerPrivateKey)
111
112 id = db.transactions.send_commit(fulfilled_token_tx)[4:] # Extract ID
113 payload = CommitTransaction(
114 id=id
115 )
116 return payload
