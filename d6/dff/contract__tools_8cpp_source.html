105int main(int argc, char** argv) {
106 if (argc < 2) {
107 printf("<cmd> -c [config]\n");
108 return 0;
109 }
110
111
112 std::string config_file;
113
114 std::string cmd;
115 std::string caller_address, contract_name, contract_path, params,
116 contract_address, func_name;
117 int c;
118 int option_index;
119 std::string client_config_file;
120 while ((c = getopt_long(argc, argv, "c:h", long_options, &option_index)) != -1) {
121 switch (c) {
122 case -1:
123 break;
124 case 'f':
125 config_file = optarg;
126 break;
127 case 'c':
128 client_config_file = optarg;
129 break;
130 case 'h':
131 ShowUsage();
132 break;
133 }
134 }
135
136 nlohmann::json js = ReadJSConfig(config_file);
137 cmd = GetValue(js, "command");
138
139 printf("client config path = %s config path = %s cmd = %s\n", client_config_file.c_str(), config_file.c_str(), cmd.c_str());
140 ResDBConfig config = GenerateResDBConfig(client_config_file);
141 config.SetClientTimeoutMs(100000);
142
143 ContractClient client(config);
144
145 if (cmd == "create_account") {
146 CreateAccount(&client);
147 } else if (cmd == "deploy") {
148
149 contract_path = GetValue(js, "contract_path");
150 contract_name = GetValue(js, "contract_name");
151 contract_address = GetValue(js, "contract_address");
152 params = GetValue(js, "init_params");
153
154 printf("contract path %s cmd %s contract name %s caller_address %s init params %s\n", contract_path.c_str(), cmd.c_str(), contract_name.c_str(), contract_address.c_str(), params.c_str());
155
156 std::vector<std::string> init_params;
157 boost::split(init_params, params, boost::is_any_of(","));
158
159 DeployContract(&client, contract_address, contract_name, contract_path,
160 init_params);
161 } else if (cmd == "execute") {
162
163 caller_address = GetValue(js, "caller_address");
164 contract_address = GetValue(js, "contract_address");
165 func_name = GetValue(js, "func_name");
166 params = GetValue(js, "params");
167
168 printf(
169 "execute\n caller address:%s\n contract address: %s\n func: %s\n "
170 "params:%s\n",
171 caller_address.c_str(), contract_address.c_str(), func_name.c_str(),
172 params.c_str());
173 std::vector<std::string> func_params;
174 boost::split(func_params, params, boost::is_any_of(","));
175
176 ExecuteContract(&client, caller_address, contract_address, func_name,
177 func_params);
178 }
179}
